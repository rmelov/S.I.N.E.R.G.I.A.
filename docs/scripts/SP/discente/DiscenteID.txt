CREATE PROCEDURE sp_DiscenteID
    @DiscenteCursoInstituicaoID UNIQUEIDENTIFIER,
    @DiscenteID UNIQUEIDENTIFIER OUTPUT
AS
BEGIN
    SET NOCOUNT ON;
    
    SELECT
        @DiscenteID = ID_DISCENTE
    FROM
        DISCENTE_CURSO_INSTITUICAO
    WHERE
        ID_DISCENTE_CURSO_INSTITUICAO = @DiscenteCursoInstituicaoID;

    IF @DiscenteID = NULL
    BEGIN
        RAISERROR(
            'Discente n√£o encontrado.',
            16,
            1
        );
        RETURN 1;
    END

    RETURN 0;
END;
